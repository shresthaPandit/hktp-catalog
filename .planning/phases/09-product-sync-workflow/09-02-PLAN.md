```javascript
const items = $input.all();
```

```javascript
const staticData = $getWorkflowStaticData('global');

const pageData = response.data ?? [];
```

**NODE 13 — Postgres: Upsert Product Batch**

- Operation: Execute Query
- Query (use Execute Query with the xmax trick for INSERT vs UPDATE counting):
```sql
INSERT INTO products (
  external_id, sku, name, description, category, subcategory,
  brand, price, image_urls, primary_image_url, specifications,
  in_stock, last_sync_at, updated_at
)
VALUES
  {{ $input.all().map(item => `(
    ${item.json.external_id ? `'${item.json.external_id.replace(/'/g, "''")}'` : 'NULL'},
    '${item.json.sku.replace(/'/g, "''")}',
    '${item.json.name.replace(/'/g, "''")}',
    ${item.json.description ? `'${item.json.description.replace(/'/g, "''")}'` : 'NULL'},
    ${item.json.category ? `'${item.json.category.replace(/'/g, "''")}'` : 'NULL'},
    ${item.json.subcategory ? `'${item.json.subcategory.replace(/'/g, "''")}'` : 'NULL'},
    ${item.json.brand ? `'${item.json.brand.replace(/'/g, "''")}'` : 'NULL'},
    ${item.json.price != null ? item.json.price : 'NULL'},
    ARRAY[${(item.json.image_urls ?? []).map(u => `'${u}'`).join(',')}],
    ${item.json.primary_image_url ? `'${item.json.primary_image_url}'` : 'NULL'},
    '${JSON.stringify(item.json.specifications ?? {})}'::jsonb,
    ${item.json.in_stock ? 'true' : 'false'},
    NOW(),
    '${item.json.updated_at}'
  )`).join(',\n  ') }}
ON CONFLICT (sku) DO UPDATE SET
  external_id = EXCLUDED.external_id,
  name = EXCLUDED.name,
  description = EXCLUDED.description,
  category = EXCLUDED.category,
  subcategory = EXCLUDED.subcategory,
  brand = EXCLUDED.brand,
  price = EXCLUDED.price,
  image_urls = EXCLUDED.image_urls,
  primary_image_url = EXCLUDED.primary_image_url,
  specifications = EXCLUDED.specifications,
  in_stock = EXCLUDED.in_stock,
  last_sync_at = NOW(),
  updated_at = EXCLUDED.updated_at,
  updated_at = NOW()
RETURNING id, sku, (xmax = 0) AS is_new_row;
```

**NODE 14 — Code: Accumulate Batch Counters**
- Type: Code
- Mode: Run Once for All Items
- JavaScript:
```javascript
const staticData = $getWorkflowStaticData('global');
const batchResults = $input.all();

// Count inserts vs updates using xmax trick (if using Execute Query with RETURNING)
// If using built-in Upsert operation, track all as synced (no insert/update distinction)
batchResults.forEach(item => {
  if (item.json.is_new_row === true) {
    staticData.totalAdded += 1;
  } else if (item.json.is_new_row === false) {
    staticData.totalUpdated += 1;
  } else {
    // Built-in Upsert mode: count all as synced
    staticData.totalSynced += 1;
  }
});

// Pass through so SplitInBatches can continue its loop
return batchResults;
```
